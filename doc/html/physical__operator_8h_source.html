56  /*
57  * A thread can choose one from the following strategies when initializing a
58  * context
59  * (1) crm_no_reuse: the thread always creates a new context whenever.
60  * (2) crm_core_sensitive: the thread first try to find and reuse a free
61  * context allocated by a thread located
62  * on the same core. If there isn't any, a new context is created.
63  * (3) crm_numa_sensitive: a free context can only be reused by a thread if
64  * the context is created by a core
65  * within the same NUMA socket.
66  * (4) crm_anyway: a free context will be reused whenever possible.
67  */
68  enum context_reuse_mode {
69  crm_no_reuse,
70  crm_core_sensitive,
71  crm_numa_sensitive,
