41  return LIKE_TRUE;

44  * In this loop, we advance by char when matching wildcards (and thus on
45  * recursive entry to this function we are properly char-synced). On other
46  * occasions it is safe to advance by byte, as the text and pattern will
47  * be in lockstep. This allows us to perform all comparisons between the
48  * text and pattern on a byte by byte basis, even for multi-byte
49  * encodings.
50  */
51  while (tlen > 0 && plen > 0)
52  {
53  if (*p == '\\')
54  {
55 // /* Next pattern byte must match literally, whatever it is */
56 // NextByte(p, plen);
57 // /* ... and there had better be one, per SQL standard */
58 // if (plen <= 0)
59 // ereport(ERROR,

63 // return LIKE_FALSE;
64  }
65  else if (*p == '%')

67  char firstpat;
68 
69  /*
70  * % processing is essentially a search for a text position at
71  * which the remainder of the text matches the remainder of the
72  * pattern, using a recursive call to check each potential match.
73  *
74  * If there are wildcards immediately following the %, we can skip
75  * over them first, using the idea that any sequence of N _'s and
76  * one or more %'s is equivalent to N _'s and one % (ie, it will
77  * match any sequence of at least N text characters). In this way
78  * we will always run the recursive search loop using a pattern
79  * fragment that begins with a literal character-to-match, thereby
80  * not recursing more than we have to.
81  */
82  NextByte(p, plen);
83 
84  while (plen > 0)
85  {
86  if (*p == '%')
87  NextByte(p, plen);
88  else if (*p == '_')
89  {
90  /* If not enough text left to match the pattern, ABORT */
91  if (tlen <= 0)
92  return LIKE_ABORT;
93  NextChar(t, tlen);
94  NextByte(p, plen);
95  }
96  else
97  break; /* Reached a non-wildcard pattern char */
98  }
99 
100  /*
101  * If we're at end of pattern, match: we have a trailing % which
102  * matches any remaining text string.
103  */
104  if (plen <= 0)
105  return LIKE_TRUE;
106 
107  /*
108  * Otherwise, scan for a text position at which we can match the
109  * rest of the pattern. The first remaining pattern char is known
110  * to be a regular or escaped literal character, so we can compare
111  * the first pattern byte to each text byte to avoid recursing
112  * more than we have to. This fact also guarantees that we don't
113  * have to consider a match to the zero-length substring at the
114  * end of the text.
115  */
116  if (*p == '\\')
117  {
118 // if (plen < 2)
119 // ereport(ERROR,
120 // (errcode(ERRCODE_INVALID_ESCAPE_SEQUENCE),
121 // errmsg("LIKE pattern must not end with escape character")));
122 // firstpat = GETCHAR(p[1]);
123  }
124  else
125  firstpat = GETCHAR(*p);
126 
127  while (tlen > 0)
128  {
129  if (GETCHAR(*t) == firstpat)
130  {
131  int matched = MatchText(t, tlen, p, plen);
132 
133  if (matched != LIKE_FALSE)
134  return matched; /* TRUE or ABORT */
135  }
136 
137  NextChar(t, tlen);
138  }
139 
140  /*
141  * End of text with no match, so no point in trying later places
142  * to start matching this pattern.
143  */
144  return LIKE_ABORT;
145  }
146  else if (*p == '_')
147  {
148  /* _ matches any single character, and we know there is one */
149  NextChar(t, tlen);
150  NextByte(p, plen);
151  continue;
152  }
153  else if (GETCHAR(*p) != GETCHAR(*t))
154  {
155  /* non-wildcard pattern char fails to match text char */
156  return LIKE_FALSE;
157  }
158 
159  /*
160  * Pattern and text match, so advance.
161  *
162  * It is safe to use NextByte instead of NextChar here, even for
163  * multi-byte character sets, because we are not following immediately
164  * after a wildcard character. If we are in the middle of a multibyte
165  * character, we must already have matched at least one byte of the
166  * character from both text and pattern; so we cannot get out-of-sync
167  * on character boundaries. And we know that no backend-legal
168  * encoding allows ASCII characters such as '%' to appear as non-first
169  * bytes of characters, so we won't mistakenly detect a new wildcard.
170  */
171  NextByte(t, tlen);
172  NextByte(p, plen);
173  }
174 
175  if (tlen > 0)
176  return LIKE_FALSE; /* end of pattern, but not of text */
177 
178  /*
179  * End of text, but perhaps not of pattern. Match iff the remaining
180  * pattern can match a zero-length string, ie, it's zero or more %'s.
181  */
182  while (plen > 0 && *p == '%')
183  NextByte(p, plen);
184  if (plen <= 0)
185  return LIKE_TRUE;
186 
187  /*
188  * End of text with no match, so no point in trying later places to start
189  * matching this pattern.
190  */
191  return LIKE_ABORT;
